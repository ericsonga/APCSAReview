14-3-13: Which of the following best explains what the method m does?

public void m(int[][]p)
{
   int height = p.length;
   for (int row = 0; row < height / 2; row++)
   {
      for (int col = 0; col <p[0].length; col++)
      {
         p[row][col] = p[height - row - 1][col];
      }
   }
}
